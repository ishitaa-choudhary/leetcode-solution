class Solution {
    public void moveZeroes(int[] arr) {
        int n=arr.length;
        int t=0;
        int[] nums=new int[n];
        for(int i=0; i<n; i++){
              if(arr[i]!=0){
                nums[t]=arr[i];
                t++;
              }
        }
        for(int i=0; i<n; i++){
            arr[i]=nums[i];
        }
    }
}
