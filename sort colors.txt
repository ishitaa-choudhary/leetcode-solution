class Solution {
    public void sortColors(int[] array) {
        int n=array.length;
        int n0=0,  n1=0;
        for(int i=0;i<n;i++) {
            if(array[i]==0) n0++;
            if(array[i]==1) n1++;
            
        }
        for(int i=0; i<n; i++) {
            if(i<n0) array[i]=0;
            else if( i<n1+n0) array[i]=1;
            else array[i]=2;
        }
            
    }
}
